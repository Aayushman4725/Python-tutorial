#While loops

#keeps executing as long as the condition holds true
c=0

while c<5
	c=c+1
	print(c)

#prints from 0 to 5 in seperate lines

#3 control statements - break, continue, pass

while c<5:
	c=c+1
	if c==3:
		break
	print(c)

#prints upto 2
while c<5:
	c=c+1
	if c==3:
		continue
	print(c)
#prints 0 to 5 excluding 3
